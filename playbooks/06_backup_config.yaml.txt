---
- name: 06 - Backup running configurations from routers and switch
  hosts: cisco-routers:cisco-switches
  gather_facts: no

  collections:
    - cisco.ios

  tasks:
    - name: Ensure local backups directory exists
      ansible.builtin.file:
        path: backups
        state: directory
        mode: "0755"
      delegate_to: localhost
      run_once: true

    - name: Collect running configuration from device
      ios_command:
        commands:
          - show running-config
      register: running_cfg

    - name: Save running configuration to local file
      ansible.builtin.copy:
        content: "{{ running_cfg.stdout[0] }}"
        dest: "backups/{{ inventory_hostname }}-running.cfg"
      delegate_to: localhost

    - name: Show backup file path
      debug:
        msg: "Backup for {{ inventory_hostname }} saved to backups/{{ inventory_hostname }}-running.cfg"
